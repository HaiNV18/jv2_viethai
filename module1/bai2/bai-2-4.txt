import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("Nhap chuoi: ");
        String chuoi = sc.nextLine();

        StringBuilder sb1 = new StringBuilder(chuoi);
        sb1.reverse();

        StringBuilder sb2 = new StringBuilder(chuoi);
        sb2.append("*");

        StringBuilder sb3 = new StringBuilder(chuoi);

        System.out.println(sb1.toString());
        System.out.println(sb2.toString());
        System.out.println(sb3.toString().replace(" ", ""));

        sc.close();
    }
}
