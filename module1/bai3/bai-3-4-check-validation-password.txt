import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("Nhap mat khau: ");
        String password = sc.nextLine();

        boolean pass = isValid(password);
        if (pass) {
            System.out.println("Password is strong");
        } else {
            System.out.println("Password is weak");
        }

        sc.close();
    }

    public static boolean isValid(String password) {
        int length = password.length();
        if (length < 8) {
            System.out.println("Password phai có 8 ky tu");
            return false;
        }

        int number = 0;
        int lowercase = 0;
        int uppercase = 0;
        int specialChar = 0;

        for (int i = 0; i < length; i++) {
            Character ch = password.charAt(i);

            if (Character.isDigit(ch)) {
                number += 1;
            } else  if (Character.isLetter(ch)) {
                System.out.println(ch);
                if (Character.isLowerCase(ch)) {
                    lowercase += 1;
                } else {
                    uppercase += 1;
                }

            } else if (!Character.isLetterOrDigit(ch) && !Character.isWhitespace(ch)) {
                specialChar += 1;
            }
        }

        if (number == 0 || lowercase == 0 || uppercase == 0 || specialChar == 0) {
            System.out.println("Mat khau phai co so, ký tự viet hoa, ký tự viet thuong và ký tự đặc biệt");
            return false;
        }

        return true;
    }
}
